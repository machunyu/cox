//! CoX UART defines the APIs based on the following common functions:
//! - \ref xUART_API_Group_NormalMode     "Configure UART as Normal Mode"
//! - \ref xUART_API_Group_IrDA           "Configure UART to Support IrDA"
//! - \ref xUART_API_Group_Modem          "UART to Support Modem Handshake"
//! - \ref xUART_API_Group_LIN            "UART to Support LIN"
//! - \ref xUART_API_Group_SmartCard	  "UART to Support SmartCard"
//! - \ref xUART_API_Group_Int            "Deal with the Interrupts"
//! .

//!
//! The CoX UART API is broken into six groups of functions.

//! -  xUARTEnable()

//! -  xUARTCharsAvail()

//! -  xUARTCharGetNonBlocking()
//! -  xUARTCharGet()

//! -  xUARTCharPut()
//! -  xUARTBusy()
//! -  xUARTRxErrorGet()
//! -  xUARTRxErrorClear()
//! .
//!

//! -  xUARTIrDADisable()
//! .
//! \n
//! \subsection xUART_API_Group_Modem 2.3 UART Modem APIs
//! -  xUARTModemControlSet()
//! -  xUARTModemControlClear()
//! -  xUARTModemControlGet()
//! -  xUARTModemStatusGet()
//! -  xUARTFlowControlSet()
//! -  xUARTFlowControlGet()
//! .
//! \n
//! \subsection xUART_API_Group_LIN 2.4 UART LIN APIs
//! -  xUARTLINConfig()
//! -  xUARTLINEnable()
//! -  xUARTLINDisable()
